# 添加的新文件无法参与编译

# ============================================================
# 自动识别用户源文件 (.c)
# CONFIGURE_DEPENDS: 每次构建前检查文件列表是否变化,
#                     若新增/删除了 .c 文件会自动触发重新 configure,
#                     避免"新加文件却没编译进去"的坑
# 注意: 这里只扫描自己新建的驱动目录,不扫描整个 Drivers,
#       避免误把 CMSIS / HAL 库的源文件重复扫入导致重复编译
# ============================================================
file(GLOB_RECURSE USER_SOURCES
    CONFIGURE_DEPENDS
    Core/Inc
    Drivers/I2C/*.c      # 自己写的 I2C 驱动
    Drivers/OLED/*.c     # 自己写的 OLED 驱动
    Drivers/SHT30/*.c    # 自己写的 SHT30 温湿度传感器驱动
)

# ============================================================
# 将上面自动识别到的 .c 源文件加入编译目标
# 后续在 Drivers/I2C、OLED、SHT30 目录下新增 .c 文件
# 会被自动扫描到,无需手动修改此处
# ============================================================
target_sources(${CMAKE_PROJECT_NAME} PRIVATE
    # Add user sources here
    ${USER_SOURCES}
)

# ============================================================
# 添加自定义头文件搜索路径 (.h)
# 每新建一个驱动目录,需要在这里手动补一行,
# 否则对应目录下的 .h 文件 #include 时会找不到
# ============================================================
target_include_directories(${CMAKE_PROJECT_NAME} PRIVATE
    # Add user defined include paths
    Core/Inc
    Drivers/I2C
    Drivers/OLED
    Drivers/SHT30
)

# sprintf() 无法支持浮点数

在项目根目录的 CMakeLists.txt 中添加/更改以下代码

```
target_link_options(${CMAKE_PROJECT_NAME} PRIVATE
    -Wl,-Map=${CMAKE_PROJECT_NAME}.map
    -u _printf_float        # ← 添加这行，开启 sprintf 浮点支持
)
```

## 没有生成HEX文件

在 CMakeLists.txt 末尾添加以下代码

```
# 生成 HEX 和 BIN 文件
add_custom_command(
  TARGET ${CMAKE_PROJECT_NAME}
  POST_BUILD
  # 生成 Intel HEX 格式
  COMMAND ${CMAKE_OBJCOPY} -O ihex $<TARGET_FILE:${CMAKE_PROJECT_NAME}> ${CMAKE_PROJECT_NAME}.hex
  # 生成 BIN 格式（可选）
  COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:${CMAKE_PROJECT_NAME}> ${CMAKE_PROJECT_NAME}.bin
  # 打印固件大小（可选）
  COMMAND ${CMAKE_SIZE} $<TARGET_FILE:${CMAKE_PROJECT_NAME}>
  COMMENT "正在生成 HEX 和 BIN 文件..."
)
```
